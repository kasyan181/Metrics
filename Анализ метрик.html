# сгруппируем данные по кварталам и чекам
s = df.groupby(['Quarter', 'Invoice']).sum()['Amount']
df2 = pd.DataFrame(s).reset_index()

# функция, выделяющая данные в рамках n стандартных отклонений
def reject_outliers(data, number_of_stdv):
    return data[abs(data - np.mean(data)) < number_of_stdv * np.std(data)]

# список значений каждого чека в первом квартале
q1_orig = np.array(sorted(df2[df2['Quarter']==1]['Amount'].to_list()))
q1_limit = q1_orig[q1_orig<8500] # 8500 - порог, по которому будут определяться аномальные значения (бирюзовая линия на графике)
q1_three_stdv = reject_outliers(q1_orig, number_of_stdv=3) # убираем данные за пределами трех стандартных отклонений
q1_two_stdv = reject_outliers(q1_orig, number_of_stdv=2) # убираем данные за пределами двух стандартных отклонений

# во втором квартале
q2_orig = np.array(sorted(df2[df2['Quarter']==2]['Amount'].to_list()))
q2_limit = q2_orig[q2_orig<8500]
q2_three_stdv = reject_outliers(q2_orig, number_of_stdv=3)
q2_two_stdv = reject_outliers(q2_orig, number_of_stdv=2)

# средние значения каждого из списков
avg_q1_orig = np.mean(q1_orig)
avg_q1_limit = np.mean(q1_limit)
avg_q1_three_stdv = np.mean(q1_three_stdv)
avg_q1_two_stdv = np.mean(q1_two_stdv)
avg_q2_orig = np.mean(q2_orig)
avg_q2_limit = np.mean(q2_limit)
avg_q2_three_stdv = np.mean(q2_three_stdv)
avg_q2_two_stdv = np.mean(q2_two_stdv)

# подсчитываем рост среднего чека
print(' Средний чек упал на', str(round((avg_q2_orig - avg_q1_orig) * -100 / avg_q1_orig))+'% без очистки данных от выбросов')
print(' Средний чек упал на', str(round((avg_q2_limit - avg_q1_limit) * -100 / avg_q1_limit))+'% после очистки данных от очевидных крупных выбросов')
print(' Средний чек упал на', str(round((avg_q2_three_stdv - avg_q1_three_stdv) * -100 / avg_q1_three_stdv))+'%, учитывая только данные в рамках трех стандартных отклонений')
print(' Средний чек упал на', str(round((avg_q2_two_stdv - avg_q1_two_stdv) * -100 / avg_q1_two_stdv))+'%, учитывая только данные в рамках двух стандартных отклонений')

Как видно из графика, метрика ARPU для нового покупателя сильно упала с января по февраль, если учитывать выбросы. Эти выбросы и могут объяснить резкое падение общего ARPU с января по февраль (как видно на предыдущем графике). В январе было некоторое количество покупателей, которые принесли исключительно большую выручку магазину. Такой тип покупателя довольно типичен для начала года, поэтому падение ARPU можно объяснить сезонностью.

Однако, следует обратить внимание, что данная метрика, очищенная от выбросов, также имеет тенденцию к снижению, пусть и в значительно более умеренном темпе. Таким образом, первое объяснение достаточно достоверно.

6) Предложите 1-2 идеи, как увеличить GMV в 3 квартале в 1,5 раза

In [62]:
GMV = np.array(df.groupby('Week').sum()['Amount']) # Выручка
N_of_uniq_cust = np.array(df.groupby('Week').nunique()['Customer ID']) # Количество уникальных покупателей
N_of_invoice = df.groupby('Week').nunique()['Invoice'] # Количество чеков
avg_invoice = GMV / np.array(N_of_invoice) # Средний чек

weeks = N_of_invoice.index

# стадтартизируем метрики, выразив их через собственные стандартные отклонения
GMV_2 = (GMV - np.mean(GMV)) / np.std(GMV)
N_of_uniq_cust_2 = (N_of_uniq_cust - np.mean(N_of_uniq_cust)) / np.std(N_of_uniq_cust)
avg_invoice_2 = (avg_invoice - np.mean(avg_invoice)) / np.std(avg_invoice)

# построим график
plt.figure(figsize=(10, 5), dpi=100)
plt.plot(weeks, GMV_2, color='black', label='GMV', linewidth = 2)
plt.plot(weeks, N_of_uniq_cust_2, color=(1,0.4,0.4), label='Количество уникальных покупателей', linestyle='dashed')
plt.plot(weeks, avg_invoice_2, color=(0.4,0.4,1), label='Средний чек', linestyle='dashed')
plt.legend()
plt.xlabel("Номер недели")
plt.title("Динамика стандартизированных метрик")
plt.show()

Как видно на графике, GMV сильнее коррелирует с количеством уникальных покупателей, нежели со средним чеком. Следовательно, для магизина, чтобы увеличить GMV, необходимо вкладывать больше ресурсов в привлечение новых и удержание старых покупателей, чем в повышение "эффективности" каждой отдельной покупки.

Из графика следует, что самое высокого значения метрика GMV достигла в первой недели июня. Предположим, что в третьем квартале удастся поддерживать среднее значение GMV на уровне этой недели. Достаточно ли будет этого, чтобы увеличить GMV в 1,5 раз?
